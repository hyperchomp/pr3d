27struct PRGLMesh *prgl_create_screen_quad(void)
28{
29 // clang-format off
30 mat4 vertices = {
31 {-1.0f, 1.0f, 0.0f}, // Top-left
32 { 1.0f, 1.0f, 0.0f}, // Top-right
33 { 1.0f, -1.0f, 0.0f}, // Bottom-right
34 {-1.0f, -1.0f, 0.0f} // Bottom-left
35 };
36 mat4x2 texture_coords = {
37 {0.0f, 1.0f}, // Top-left
38 {1.0f, 1.0f}, // Top-right
39 {1.0f, 0.0f}, // Bottom-right
40 {0.0f, 0.0f} // Bottom-left
41 };
42
43 float combined_data[20] = {
44 vertices[0][0], vertices[0][1], vertices[0][2],
45 texture_coords[0][0], texture_coords[0][1],
46 vertices[1][0], vertices[1][1], vertices[1][2],
47 texture_coords[1][0], texture_coords[1][1],
48 vertices[2][0], vertices[2][1], vertices[2][2],
49 texture_coords[2][0], texture_coords[2][1],
50 vertices[3][0], vertices[3][1], vertices[3][2],
51 texture_coords[3][0], texture_coords[3][1]
52 };
53 // clang-format on
54
55 // EBO stops us from needing overlapping vertices, but we need to tell
56 // OpenGL the order to go over the existing ones again to create enough
57 // triangles to create our mesh (in this case 2 triangles, 6 indices)
58 unsigned int indices[] = {
59 0, 1, 3, // First triangle
60 1, 2, 3 // Second triangle
61 };
62
63 // Create a vertex buffer object and vertex array object, the VBO is to
64 // generate the initial data, the VAO is so we can re-use it later
65 unsigned int vbo;
66 unsigned int vao;
67 unsigned int ebo;
68 glGenBuffers(1, &vbo);
69 glGenBuffers(1, &ebo);
70 glGenVertexArrays(1, &vao);
71
72 // Bind VAO, then bind and set buffers, then configure the vertex attributes
73 glBindVertexArray(vao);
74 glBindBuffer(GL_ARRAY_BUFFER, vbo);
75 glBufferData(
76 GL_ARRAY_BUFFER, sizeof(combined_data), combined_data, GL_STATIC_DRAW
77 );
78 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, ebo);
79 glBufferData(
80 GL_ELEMENT_ARRAY_BUFFER, sizeof(indices), indices, GL_STATIC_DRAW
81 );
82
83 // Tell OpenGL how to interpret the vertex data
84 glVertexAttribPointer(
85 0, 3, GL_FLOAT, GL_FALSE, 5 * sizeof(float), (void *)0
86 );
87 glEnableVertexAttribArray(0);
88
89 // Tell OpenGL how to interpret the texture coordinate data
90 glVertexAttribPointer(
91 2, 2, GL_FLOAT, GL_FALSE, 5 * sizeof(float), (void *)(3 * sizeof(float))
92 );
93 glEnableVertexAttribArray(2);
94
95 struct PRGLMesh *mesh_pointer = malloc(sizeof(struct PRGLMesh));
96 if (mesh_pointer == NULL)
97 {
98 printf("prgl_create_screen_quad: Error allocating mesh "
99 "pointer memory!");
100 }
101
102 // Texture zero'd, must be set after using load_texture()
103 *mesh_pointer = (struct PRGLMesh){
104 .num_vertices = ARR_LEN(indices),
105 .vao = vao,
106 .vbo = vbo,
107 .ebo = ebo,
108 .texture = 0,
109 };
110 return mesh_pointer;
111}
