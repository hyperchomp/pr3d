463struct PR3DRenderTexture pr3d_create_render_texture(void)
464{
465 // Create a framebuffer object
466 unsigned int fbo;
467 glGenFramebuffers(1, &fbo);
468 glBindFramebuffer(GL_FRAMEBUFFER, fbo);
469
470 // Create the texture for rendering to
471 unsigned int render_texture;
472 glGenTextures(1, &render_texture);
473 glBindTexture(GL_TEXTURE_2D, render_texture);
474 glTexImage2D(
475 GL_TEXTURE_2D, 0, GL_RGB, PR3D_RENDER_RESOLUTION[0],
476 PR3D_RENDER_RESOLUTION[1], 0, GL_RGB, GL_UNSIGNED_BYTE, NULL
477 );
478 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
479 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
480
481 // Attach the texture to the FBO
482 glFramebufferTexture2D(
483 GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, render_texture, 0
484 );
485
486 // Attach a combined depth and stencil render buffer object
487 unsigned int rbo;
488 glGenRenderbuffers(1, &rbo);
489 glBindRenderbuffer(GL_RENDERBUFFER, rbo);
490 glRenderbufferStorage(
491 GL_RENDERBUFFER, GL_DEPTH24_STENCIL8, PR3D_RENDER_RESOLUTION[0],
492 PR3D_RENDER_RESOLUTION[1]
493 );
494 glFramebufferRenderbuffer(
495 GL_FRAMEBUFFER, GL_DEPTH_STENCIL_ATTACHMENT, GL_RENDERBUFFER, rbo
496 );
497
498 if (glCheckFramebufferStatus(GL_FRAMEBUFFER) != GL_FRAMEBUFFER_COMPLETE)
499 {
500 fprintf(
501 stderr, "pr3d_create_render_texture: Tried to create framebuffer "
502 "but framebuffer is not complete!\n"
503 );
504 exit(EXIT_FAILURE);
505 }
