31unsigned int pr3d_create_shader(
32 const char *const vertex_source, const char *const frag_source
33)
34{
35 // Create a basic vertex shader
36 unsigned int vertex_shader =
37 pr3d_compile_shader(GL_VERTEX_SHADER, vertex_source);
38
39 // Create a fragment shader for color
40 unsigned int fragment_shader =
41 pr3d_compile_shader(GL_FRAGMENT_SHADER, frag_source);
42
43 // Use a shader program to link the shaders together
44 unsigned int shaders[] = {vertex_shader, fragment_shader};
45 unsigned int shader_program =
46 pr3d_create_shader_program(shaders, ARR_LEN(shaders));
47
48 // Delete the shader objects, we're done with them
49 glDeleteShader(vertex_shader);
50 glDeleteShader(fragment_shader);
51
52 return shader_program;
53}

166static unsigned int

168{
169 unsigned int shader_program = glCreateProgram();
170 for (size_t i = 0; i < length; i++)
