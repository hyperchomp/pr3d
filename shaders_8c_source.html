36unsigned int prgl_create_shader(
37 const char *const vertex_source, const char *const frag_source
38)
39{
40 // Create a basic vertex shader
41 unsigned int vertex_shader =
42 prgl_compile_shader(GL_VERTEX_SHADER, vertex_source);
43
44 // Create a fragment shader for color
45 unsigned int fragment_shader =
46 prgl_compile_shader(GL_FRAGMENT_SHADER, frag_source);
47
48 // Use a shader program to link the shaders together
49 unsigned int shaders[] = {vertex_shader, fragment_shader};
50 unsigned int shader_program =
51 prgl_create_shader_program(shaders, ARR_LEN(shaders));
52
53 // Delete the shader objects, we're done with them
54 glDeleteShader(vertex_shader);
55 glDeleteShader(fragment_shader);

57 return shader_program;
